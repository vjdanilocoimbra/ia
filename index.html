""" N.E.X.U.S. Voice Service — Modulo 1 (Base) Servidor basico: texto -> IA -> resposta """ import os, json, asyncio from fastapi import FastAPI from fastapi.middleware.cors import CORSMiddleware from pydantic import BaseModel from openai import AsyncOpenAI import uvicorn _nexus_keys = {} try: with open(os.path.expanduser("~/.nanobot/nexus_settings.json"), "r", encoding="utf-8") as f: _nexus_keys = json.load(f).get("keys", {}) except Exception: pass OPENAI_KEY = os.getenv("OPENAI_API_KEY", "") or _nexus_keys.get("openai_api_key", "") AI_MODEL = "gpt-4o-mini" if not OPENAI_KEY: print("[ERRO] Configure openai_api_key em ~/.nanobot/nexus_settings.json") exit(1) client = AsyncOpenAI(api_key=OPENAI_KEY) app = FastAPI(title="N.E.X.U.S. Voice Service") app.add_middleware(CORSMiddleware, allow_origins=["*"], allow_methods=["*"], allow_headers=["*"]) SYSTEM_PROMPT = """Voce e um assistente pessoal de IA. Responda em portugues brasileiro, de forma concisa e direta. Trate o usuario como "senhor". Maximo 2-3 frases por resposta.""" _history = [{"role": "system", "content": SYSTEM_PROMPT}] class ChatRequest(BaseModel): text: str @app.post("/chat") async def chat(body: ChatRequest): _history.append({"role": "user", "content": body.text}) response = await client.chat.completions.create( model=AI_MODEL, messages=_history[-20:], temperature=0.3, max_tokens=500, ) reply = response.choices[0].message.content _history.append({"role": "assistant", "content": reply}) return {"reply": reply, "model": AI_MODEL} @app.get("/status") async def status(): return {"status": "online", "model": AI_MODEL, "history": len(_history)} if __name__ == "__main__": print("=" * 50) print(" N.E.X.U.S. Voice Service — Modulo 1") print("=" * 50) uvicorn.run("server_basico:app", host="0.0.0.0", port=5057, log_level="info")
